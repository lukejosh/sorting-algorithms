\begin{algorithmic}
\Function{InsertionSort}{$A[0..n-1$}
\For{$i \leftarrow 1$ \bf{to} $n - 1$}
    \State{$v \leftarrow A[i]$}
    \State{$j \leftarrow i - 1$}
    \While{$j \geq 0$ \bf{and} $A[j] > v$}
        \State{$A[j + 1] \leftarrow A[j]$}
        \State{$j \leftarrow j - 1$}
    \EndWhile
    \State{$A[j + 1] \leftarrow v$}
\EndFor
\EndFunction
\end{algorithmic}
